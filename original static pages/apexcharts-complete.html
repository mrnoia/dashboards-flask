    <title>My Chart</title>

        new ApexCharts(document.querySelector("#myChart"), options).render();
    </script>
</body>
</html>

Interactive & Drillable Dashboard Examples

Complete examples showing how to build production-ready dashboards with Bootstrap and ApexCharts.

üí∞ Sales Dashboard Example

Features metric cards, interactive charts, and drilldown functionality.

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Sales Dashboard</title>
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.min.css" rel="stylesheet">
    <style>
        .metric-card {
            background: linear-gradient(135deg, #667eea 0%, #764ba2 100%);
            color: white;
            border-radius: 15px;
            padding: 20px;
            text-align: center;
            transition: transform 0.3s ease;
        }
        .metric-card:hover {
            transform: translateY(-5px);
        }
        .chart-container {
            min-height: 350px;
        }
    </style>
</head>
<body>
    <div class="container-fluid py-4">
        <h1 class="mb-4">Sales Dashboard</h1>
        
        <!-- Metrics Row -->
        <div class="row mb-4">
            <div class="col-md-3">
                <div class="metric-card" onclick="drillToCategory('Electronics')">
                    <h6>Electronics Revenue</h6>
                    <h3>$45,000</h3>
                    <small>Click to drill down ‚Üí</small>
                </div>
            </div>
        </div>
        
        <!-- Chart Row -->
        <div class="row">
            <div class="col-12">
                <div class="card p-4">
                    <h5>Revenue by Category</h5>
                    <div id="revenue-chart"></div>
                </div>
            </div>
        </div>
    </div>
    
    <script src="https://cdn.jsdelivr.net/npm/apexcharts"></script>
    <script>
        // Chart data
        const salesData = {
            categories: ['Electronics', 'Clothing', 'Home & Garden', 'Sports'],
            revenue: [45000, 32000, 28000, 18000]
        };
        
        // Chart configuration
        const options = {
            chart: {
                type: 'bar',
                height: 350,
                events: {
                    dataPointSelection: function(event, chartContext, config) {
                        drillToCategory(salesData.categories[config.dataPointIndex]);
                    }
                }
            },
            series: [{
                name: 'Revenue',
                data: salesData.revenue
            }],
            colors: ['#3498db', '#e74c3c', '#f39c12', '#2ecc71']
        };
        
        // Create chart
        const chart = new ApexCharts(document.querySelector("#revenue-chart"), options);
        chart.render();
        
        // Drilldown function
        function drillToCategory(category) {
            alert(`Drilling down to ${category} details...`);
            // Load detailed data for selected category
        }
    </script>
</body>
</html>
